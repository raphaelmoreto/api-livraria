CREATE TABLE autor (

	nome VARCHAR (100) NOT NULL,
	status_autor BIT DEFAULT 1

	id INT IDENTITY PRIMARY KEY,
	titulo VARCHAR(100) NOT NULL,
	ano_publicacao DATE NOT NULL,
	fk_autor INT
)
